def force_attributes(new_attributes, options = {})
  new_attributes = @attributes.merge(new_attributes) if options[:merge]
  @attributes = new_attributes
  
  if options[:clear_caches] != false
    @aggregation_cache = {}
    @association_cache = {}
    @attributes_cache = {}
  end
end
